// Escuchar cuando el pasajero reserva un asiento _s.channel('alertas-chofer').on('postgres_changes', {event:'UPDATE', schema:'public', table:'estado_movil'}, payload => { const d = payload.new; // 1. Actualiza tu contador de pasajeros arriba a la derecha document.getElementById('v-pax').innerText = 4 - d.asientos_libres; // 2. Si los asientos bajaron, avisarte visualmente if (d.asientos_libres < 4) { // Hace que el panel de pasajeros parpadee en verde para avisarte const panel = document.querySelector('.pax-panel'); panel.style.background = "#00ff88"; setTimeout(() => { panel.style.background = "rgba(0,0,0,0.8)"; }, 2000); // Si el celu lo permite, vibra if (navigator.vibrate) navigator.vibrate(500); } // 3. Pintar los botones de asientos automáticamente actualizarBotonesChofer(d.asientos_libres); }).subscribe(); function actualizarBotonesChofer(libres) { const ocupados = 4 - libres; const botones = document.querySelectorAll('.btn-pax'); botones.forEach((btn, index) => { if (index < ocupados) { btn.classList.add('active'); // Se pone rojo } else { btn.classList.remove('active'); // Se pone gris } }); }
